    ```typescript
    @Get('reports/executive')
    async getExecutiveSummary(
      @Request() req: any,
      @Query('year') year: string,
      @Query('month') month: string,
    ) {
      const { role, companyId } = req.user;
      // Accessible by executive, admin, owner (executive has read-only access per EXEC-05)
      if (!['executive', 'admin', 'owner'].includes(role)) {
        throw new ForbiddenException('Insufficient permissions');
      }
      const y = parseInt(year) || new Date().getFullYear();
      const m = parseInt(month) || new Date().getMonth() + 1;
      return this.attendanceService.getExecutiveSummary(companyId, y, m);
    }

    @Get('reports/monthly')
    async getMonthlyReport(
      @Request() req: any,
      @Query('year') year: string,
      @Query('month') month: string,
    ) {
      const { role, companyId, userId } = req.user;
      if (!['admin', 'owner', 'manager'].includes(role)) {
        throw new ForbiddenException('Insufficient permissions');
      }
      const y = parseInt(year) || new Date().getFullYear();
      const m = parseInt(month) || new Date().getMonth() + 1;
      const managerId = role === 'manager' ? userId : undefined;
      return this.attendanceService.getMonthlyReport(companyId, y, m, managerId);
    }

    @Get('export/csv')
    async exportCsv(
      @Request() req: any,
      @Query('year') year: string,
      @Query('month') month: string,
      @Res() res: Response,
    ) {
      const { role, companyId, userId } = req.user;
      if (!['admin', 'owner', 'manager'].includes(role)) {
        throw new ForbiddenException('Insufficient permissions');
      }
      const y = parseInt(year) || new Date().getFullYear();
      const m = parseInt(month) || new Date().getMonth() + 1;
      const managerId = role === 'manager' ? userId : undefined;
      const csv = await this.attendanceService.exportCsv(companyId, y, m, managerId);
      res.setHeader('Content-Type', 'text/csv');
      res.setHeader('Content-Disposition', `attachment; filename="attendance-${y}-${String(m).padStart(2, '0')}.csv"`);
      res.send(csv);
    }
    ```

    Note: When using `@Res()` with NestJS, the response is handled manually — do NOT return a value from that method (use `res.send()` instead).

    Note: The `reports/team-summary` route was added in plan 05-01. Place new routes alongside it — order: `reports/team-summary`, `reports/executive`, `reports/monthly`, `export/csv`.
