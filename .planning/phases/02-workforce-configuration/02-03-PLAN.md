    - endTime: string (@IsString @Matches(/^\d{2}:\d{2}$/))

    Create backend/src/shifts/dto/update-shift.dto.ts:

    Create backend/src/shifts/shifts.service.ts injecting SupabaseService:
    - listShifts(companyId: string): queries shifts WHERE company_id=companyId ORDER BY name ASC, returns Shift[]
    - createShift(companyId: string, dto: CreateShiftDto):
      1. Insert into shifts: { company_id: companyId, name: dto.name, start_time: dto.startTime, end_time: dto.endTime, grace_period_minutes: dto.gracePeriodMinutes }
      2. On error throw ConflictException
      3. Returns created shift record
    - updateShift(companyId: string, shiftId: string, dto: UpdateShiftDto):
      1. Build updateData from only defined fields (name → name, startTime → start_time, endTime → end_time, gracePeriodMinutes → grace_period_minutes)
      2. Update shifts WHERE id=shiftId AND company_id=companyId
      3. If no rows returned throw NotFoundException
      4. Returns updated shift record

    Create backend/src/shifts/shifts.controller.ts:
    - @Controller('shifts') @UseGuards(JwtAuthGuard)
    - GET /shifts → listShifts(req.user.companyId)
    - POST /shifts → createShift(req.user.companyId, dto)
    - PATCH /shifts/:id → updateShift(req.user.companyId, params.id, dto)

    - Controllers: [ShiftsController]
