    - CREATE TABLE IF NOT EXISTS employee_shifts (
        id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
        company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
        user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
        shift_id UUID NOT NULL REFERENCES shifts(id) ON DELETE CASCADE,
        effective_date DATE NOT NULL,
        created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
        UNIQUE (user_id, effective_date)  -- prevents duplicate assignments on same date
      );
    - CREATE INDEX IF NOT EXISTS idx_employee_shifts_company_id ON employee_shifts(company_id);
    - CREATE INDEX IF NOT EXISTS idx_employee_shifts_user_id ON employee_shifts(user_id);

    Create backend/src/users/dto/create-user.dto.ts:
    - fullName: string (@IsString @IsNotEmpty)
    - email: string (@IsEmail)
    - password: string (@IsString @MinLength(8))
    - role: string (@IsIn(['owner','admin','manager','employee','executive']))
    - managerId?: string (@IsUUID @IsOptional) — only relevant when role='employee'

    Create backend/src/users/dto/update-user.dto.ts:
    - role?: string (@IsIn(['owner','admin','manager','employee','executive']) @IsOptional)
    - managerId?: string (@IsUUID @IsOptional)

    Create backend/src/users/users.service.ts injecting SupabaseService:
    - listUsers(companyId: string): queries users WHERE company_id=companyId, orders by created_at ASC, returns array
    - createUser(companyId: string, dto: CreateUserDto):
      1. Creates Supabase auth user via client.auth.admin.createUser({ email, password, email_confirm: true, app_metadata: { company_id: companyId, role: dto.role } })
      2. On authError throw ConflictException
      3. Inserts into public.users: { id: authData.user.id, company_id: companyId, full_name: dto.fullName, email: dto.email, role: dto.role, manager_id: dto.managerId ?? null }
      4. On userError: rollback (delete auth user), throw ConflictException
      5. Returns created user record
    - updateUser(companyId: string, userId: string, dto: UpdateUserDto):
      1. Builds updateData object from only defined fields (role, manager_id)
      2. Updates users WHERE id=userId AND company_id=companyId (tenant-scoped)
      3. On error throw BadRequestException
      4. Returns updated user record; if no rows returned throw NotFoundException
    - setUserStatus(companyId: string, userId: string, isActive: boolean):
      1. Updates users SET is_active=isActive WHERE id=userId AND company_id=companyId
      2. Also updates Supabase auth user ban_duration: isActive ? 'none' : '100 years' via client.auth.admin.updateUserById(userId, { ban_duration: isActive ? 'none' : '876000h' })
      3. Returns updated record

    Create backend/src/users/users.controller.ts:
    - @Controller('users') @UseGuards(JwtAuthGuard)
    - GET /users → listUsers(req.user.companyId)
    - POST /users → createUser(req.user.companyId, dto)
    - PATCH /users/:id → updateUser(req.user.companyId, params.id, dto) — updates role and/or managerId
    - PATCH /users/:id/status → body: { isActive: boolean } → setUserStatus(req.user.companyId, params.id, body.isActive)
    - All routes use @Request() req to extract companyId and userId from JWT payload
    - Use @Param('id') for :id
