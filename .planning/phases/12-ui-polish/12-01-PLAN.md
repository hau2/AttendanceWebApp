    ```tsx
    import { CheckCircle, Clock, LogOut, AlertCircle, XCircle, Laptop } from 'lucide-react';

    export function StatusBadge({ status, missingCheckout }: { status: string | null; missingCheckout?: boolean }) {
      if (missingCheckout) {
        return (
          <span className="inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-xs font-medium bg-gray-100 text-gray-600">
            <AlertCircle className="w-3 h-3" />
            Missing Checkout
          </span>
        );
      }
      // ... map status to icon + colors
      if (!status || !config) return null;
      const { icon: Icon, cls, label } = config;
      return (
        <span className={`inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-xs font-medium ${cls}`}>
          <Icon className="w-3 h-3" />
          {label}
        </span>
      );
    }

    export function RemoteBadge() {
      return (
        <span className="inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-xs font-medium bg-blue-100 text-blue-700">
          <Laptop className="w-3 h-3" />
          Remote
        </span>
      );
    }
    ```
