const API_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:3001';

export interface AttendanceRecord {
  id: string;
  company_id: string;
  user_id: string;
  work_date: string;
  check_in_at: string | null;
  check_in_photo_url: string | null;
  check_in_ip: string | null;
  check_in_status: 'on-time' | 'within-grace' | 'late' | null;
  minutes_late: number;
  late_reason: string | null;
  check_in_ip_within_allowlist: boolean | null;
  check_out_at: string | null;
  check_out_photo_url: string | null;
  check_out_ip: string | null;
  check_out_status: 'on-time' | 'early' | null;
  minutes_early: number;
  early_note: string | null;
  check_out_ip_within_allowlist: boolean | null;
  missing_checkout: boolean;
  source: 'employee' | 'system' | 'admin';
  created_at: string;
  updated_at: string;
}

export interface PhotoUploadUrlResponse {
  signedUrl: string;
  permanentUrl: string;
  path: string;
  expiresIn: number;
}

function getToken(): string {
  return localStorage.getItem('access_token') || '';
}

export async function getPhotoUploadUrl(): Promise<PhotoUploadUrlResponse> {
  const res = await fetch(`${API_URL}/attendance/photo-upload-url`, {
    method: 'POST',
    headers: { Authorization: `Bearer ${getToken()}` },
  });
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    throw new Error(err.message || 'Failed to get photo upload URL');
  }
  return res.json();
}

export async function uploadPhotoBlob(signedUrl: string, blob: Blob): Promise<void> {
  const res = await fetch(signedUrl, {
    method: 'PUT',
    body: blob,
    headers: { 'Content-Type': 'image/jpeg' },
  });
  if (!res.ok) {
    throw new Error('Photo upload failed');
  }
}

export async function checkIn(data: {
  photo_url?: string;
  late_reason?: string;
}): Promise<AttendanceRecord> {
  const res = await fetch(`${API_URL}/attendance/check-in`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${getToken()}`,
    },
    body: JSON.stringify(data),
  });
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    throw new Error(err.message || 'Check-in failed');
  }
  return res.json();
}

export async function checkOut(data: {
  photo_url?: string;
  early_note?: string;
}): Promise<AttendanceRecord> {
  const res = await fetch(`${API_URL}/attendance/check-out`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${getToken()}`,
    },
    body: JSON.stringify(data),
  });
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    throw new Error(err.message || 'Check-out failed');
  }
  return res.json();
}

export async function getTodayRecord(): Promise<AttendanceRecord | null> {
  const today = new Date().toLocaleDateString('en-CA'); // YYYY-MM-DD
  const res = await fetch(
    `${API_URL}/attendance/history?year=${new Date().getFullYear()}&month=${new Date().getMonth() + 1}`,
    { headers: { Authorization: `Bearer ${getToken()}` } },
  );
  if (!res.ok) return null;
  const records: AttendanceRecord[] = await res.json();
  return records.find((r) => r.work_date === today) ?? null;
}

export async function getHistory(year: number, month: number): Promise<AttendanceRecord[]> {
  const res = await fetch(`${API_URL}/attendance/history?year=${year}&month=${month}`, {
    headers: { Authorization: `Bearer ${getToken()}` },
  });
  if (!res.ok) return [];
  return res.json();
}

Create `frontend/src/app/(app)/dashboard/components/CheckInOutCard.tsx`:

This is a 'use client' component managing the full check-in/out flow. Implement the following state machine:

States:
- `idle` — showing CHECK-IN button (or CHECK-OUT if already checked in)
- `camera-open` — camera stream active, live preview shown, Capture button visible
- `photo-preview` — photo captured, thumbnail shown, Submit button visible (+ reason/note field if needed)
- `submitting` — loading spinner
- `error` — error message shown with retry

On mount (useEffect): call getTodayRecord() to check if already checked in or checked out today. Set initial state accordingly.

Camera logic:
- onClick CHECK-IN or CHECK-OUT: call `navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } })` for mobile (environment = rear camera) — if that fails, fallback to `{ video: true }`.
- Show `<video ref={videoRef} autoPlay playsInline />` with stream.
- Capture: draw video frame to a canvas (videoRef.current.videoWidth x videoRef.current.videoHeight), call `canvas.toBlob((blob) => ..., 'image/jpeg', 0.85)`.
- Stop all tracks after capture: `stream.getTracks().forEach(t => t.stop())`.
- Show captured photo as `<img src={URL.createObjectURL(blob)} />` thumbnail.

Late reason / early note:
- The component does NOT know status before submission (backend classifies). Instead:
  - Attempt submission without reason/note first.
  - If backend returns HTTP 400 with message containing "requires a reason" → show late_reason text field in UI, require user to fill it, re-submit.
  - If backend returns HTTP 400 with message containing "requires a note" → show early_note text field, re-submit.
  - This avoids duplicating classification logic in the frontend.

Submit flow:
1. Call getPhotoUploadUrl() → { signedUrl, permanentUrl }
2. Call uploadPhotoBlob(signedUrl, capturedBlob)
3. Call checkIn({ photo_url: permanentUrl, late_reason? }) or checkOut({ photo_url: permanentUrl, early_note? })
4. On success: update today's record state, change button to CHECK-OUT / show "Checked out" message.
5. On error: show error message, return to photo-preview state so user can retry.

UI layout (Tailwind):
- Card: `bg-white rounded-xl shadow-md p-8 max-w-sm mx-auto text-center`
- CHECK-IN button: `w-full py-4 px-6 bg-green-600 text-white text-xl font-bold rounded-xl hover:bg-green-700 transition-colors`
- CHECK-OUT button: `w-full py-4 px-6 bg-red-600 text-white text-xl font-bold rounded-xl hover:bg-red-700 transition-colors`
- Show check-in time if already checked in: `<p className="text-sm text-gray-500">Checked in at {time}</p>`
- Camera container: `relative w-full aspect-video bg-black rounded-lg overflow-hidden`
- Capture button: `w-full py-3 bg-blue-600 text-white font-semibold rounded-lg mt-3`
- Error message: `text-red-600 text-sm mt-2`
- No shift message: `text-gray-500 text-sm`

No shift case: If `record === null` on mount AND user tries to check in, backend will return with status='on-time' (no shift = no classification). The "no shift assigned" message is only shown if the backend returns a specific error. Keep it simple — let backend handle it.

IP blocked case: If backend returns 403, show "Check-in blocked: your IP address is not in the company allowlist" in red.

'use client';
import { useEffect, useState } from 'react';
import { getStoredUser, AuthUser } from '@/lib/api/auth';
import { CheckInOutCard } from './components/CheckInOutCard';

export default function DashboardPage() {
  const [user, setUser] = useState<AuthUser | null>(null);

  useEffect(() => {
    setUser(getStoredUser());
  }, []);

  if (!user) return null;

  // Admins, executives, and owners see a different dashboard (Phase 5)
  const isAttendanceParticipant = ['employee', 'manager'].includes(user.role);

  return (
    <div className="max-w-2xl mx-auto">
      <div className="mb-6">
        <h1 className="text-2xl font-bold text-gray-900">Welcome, {user.full_name}</h1>
        <p className="text-gray-500 text-sm mt-1">{new Date().toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' })}</p>
      </div>

      {isAttendanceParticipant ? (
        <CheckInOutCard />
      ) : (
        <div className="bg-white rounded-xl shadow-md p-8 text-center">
          <p className="text-gray-500">Attendance tracking dashboard coming in Phase 5.</p>
          <p className="text-gray-400 text-sm mt-2">Your role: {user.role}</p>
        </div>
      )}
    </div>
  );
}
