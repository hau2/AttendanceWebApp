```sql
-- Migration 004: Attendance Records
-- Run via Supabase SQL editor

CREATE TABLE IF NOT EXISTS attendance_records (
  id                 UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  company_id         UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
  user_id            UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  work_date          DATE NOT NULL,

  -- Check-in fields
  check_in_at        TIMESTAMPTZ,
  check_in_photo_url TEXT,
  check_in_ip        TEXT,
  check_in_status    TEXT CHECK (check_in_status IN ('on-time', 'within-grace', 'late')),
  minutes_late       INT NOT NULL DEFAULT 0,
  late_reason        TEXT,
  check_in_ip_within_allowlist BOOLEAN,

  -- Check-out fields
  check_out_at       TIMESTAMPTZ,
  check_out_photo_url TEXT,
  check_out_ip       TEXT,
  check_out_status   TEXT CHECK (check_out_status IN ('on-time', 'early')),
  minutes_early      INT NOT NULL DEFAULT 0,
  early_note         TEXT,
  check_out_ip_within_allowlist BOOLEAN,

  -- System status
  missing_checkout   BOOLEAN NOT NULL DEFAULT FALSE,
  source             TEXT NOT NULL DEFAULT 'employee' CHECK (source IN ('employee', 'system', 'admin')),

  created_at         TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at         TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  -- One record per employee per work_date
  UNIQUE (user_id, work_date)
);

CREATE INDEX IF NOT EXISTS idx_attendance_company_id ON attendance_records(company_id);
CREATE INDEX IF NOT EXISTS idx_attendance_user_id ON attendance_records(user_id);
CREATE INDEX IF NOT EXISTS idx_attendance_work_date ON attendance_records(work_date);
CREATE INDEX IF NOT EXISTS idx_attendance_company_work_date ON attendance_records(company_id, work_date);
```

Key logic to implement:

**getWorkDate(timezone: string): string** — returns today's date string (YYYY-MM-DD) in the company's timezone. Use `new Date().toLocaleDateString('en-CA', { timeZone: timezone })` which produces YYYY-MM-DD format natively (no extra library needed).

**classifyCheckIn(checkInAt: Date, shift: ShiftAssignmentWithShift, timezone: string):**
- Parse shift.start_time as HH:MM. Construct shift start as a Date for today in company timezone.
- `const [sh, sm] = shift.shifts.start_time.split(':').map(Number)`
- Build shiftStartMs: get today's date string in timezone, then `new Date(${workDate}T${String(sh).padStart(2,'0')}:${String(sm).padStart(2,'0')}:00)` — treat this as local timezone-naive then compare as timestamps in UTC.
- **Simpler approach for v1:** Compare minutes since midnight in company timezone for both check-in and shift start. Use `checkInAt.toLocaleTimeString('en-US', { hour12: false, hour: '2-digit', minute: '2-digit', timeZone: timezone })` → parse as minutes. Compare with shift start/end minutes.
- If checkInMinutes <= shiftStartMinutes → 'on-time', minutes_late = 0
- If checkInMinutes <= shiftStartMinutes + grace_period_minutes → 'within-grace', minutes_late = checkInMinutes - shiftStartMinutes (could be negative if within grace — use 0 or actual offset as appropriate; set to minutes past shift start, max(0, checkInMinutes - shiftStartMinutes))
- If checkInMinutes > shiftStartMinutes + grace_period_minutes → 'late', minutes_late = checkInMinutes - shiftStartMinutes

**classifyCheckOut(checkOutAt: Date, shift: ShiftAssignmentWithShift, timezone: string):**
- Parse shift.shifts.end_time similarly.
- If checkOutMinutes >= shiftEndMinutes → 'on-time', minutes_early = 0
- If checkOutMinutes < shiftEndMinutes → 'early', minutes_early = shiftEndMinutes - checkOutMinutes

**checkIP(companyId: string, callerIp: string):** Fetch company's ip_mode and ip_allowlist from companies table. Return `{ withinAllowlist: boolean, blocked: boolean }`. If allowlist is empty array, treat as "no restriction" — withinAllowlist=true, blocked=false regardless of mode. Otherwise check if callerIp is in allowlist. If enforce-block and not in allowlist → blocked=true. If log-only → blocked=false regardless.

**checkIn(companyId, userId, ip, dto):**
1. Fetch company timezone and IP settings.
2. IP check — throw ForbiddenException if blocked.
3. getWorkDate using company timezone.
4. Check for existing record for (user_id, work_date) — throw ConflictException('Already checked in today') if found.
5. Get active shift via ShiftAssignmentsService.getActiveShift. If null, use status='on-time', minutes_late=0 (no shift = cannot classify; proceed without blocking).
6. Classify check-in.
7. If check_in_status='late' and no late_reason → throw BadRequestException('Late check-in requires a reason').
8. Insert attendance_records row. Return the created record.

**checkOut(companyId, userId, ip, dto):**
1. Fetch company settings.
2. IP check — throw ForbiddenException if blocked.
3. getWorkDate.
4. Fetch today's record for user — throw NotFoundException('No check-in found for today') if missing.
5. If record.check_out_at is not null → throw ConflictException('Already checked out today').
6. Get active shift. Classify check-out (if no shift, status='on-time').
7. If check_out_status='early' and no early_note → throw BadRequestException('Early check-out requires a note').
8. Update record with check-out fields. Return updated record.

**getHistory(companyId, userId, year, month):**
- Query attendance_records WHERE company_id AND user_id AND date_trunc('month', work_date) = given month/year.
- Return array ordered by work_date DESC.

```typescript
@Controller('attendance')
@UseGuards(JwtAuthGuard)
export class AttendanceController {
  // Inject AttendanceService

  @Post('check-in')
  async checkIn(@Request() req, @Body() dto: CheckInDto) {
    const ip = req.headers['x-forwarded-for']?.split(',')[0]?.trim()
              || req.socket?.remoteAddress
              || 'unknown';
    return this.attendanceService.checkIn(req.user.companyId, req.user.userId, ip, dto);
  }

  @Post('check-out')
  async checkOut(@Request() req, @Body() dto: CheckOutDto) {
    const ip = req.headers['x-forwarded-for']?.split(',')[0]?.trim()
              || req.socket?.remoteAddress
              || 'unknown';
    return this.attendanceService.checkOut(req.user.companyId, req.user.userId, ip, dto);
  }

  @Get('history')
  async getHistory(@Request() req, @Query('year') year: string, @Query('month') month: string) {
    const y = parseInt(year) || new Date().getFullYear();
    const m = parseInt(month) || new Date().getMonth() + 1;
    return this.attendanceService.getHistory(req.user.companyId, req.user.userId, y, m);
  }

  // Admin/Manager: list records for all users in company (filtered by date range)
  @Get('records')
  async listRecords(
    @Request() req,
    @Query('year') year: string,
    @Query('month') month: string,
    @Query('userId') userId?: string,
  ) {
    // Only admin, owner, manager roles can list all records
    const { role, companyId } = req.user;
    if (!['admin', 'owner', 'manager'].includes(role)) {
      throw new ForbiddenException('Insufficient permissions');
    }
    const y = parseInt(year) || new Date().getFullYear();
    const m = parseInt(month) || new Date().getMonth() + 1;
    return this.attendanceService.listRecords(companyId, y, m, userId);
  }
}
```
